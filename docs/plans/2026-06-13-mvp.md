# FormalizationFlow MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A plasTeX plugin that lets a mathematician write one `.tex` file and build it as PDF (pdflatex), HTML (plastex), and sorry-filled Lean 4 (plastex --renderer lean4).

**Architecture:** The Python package `formalization_flow` acts as a plasTeX plugin providing two things: a LaTeX Package (`\usepackage{formalizationflow}`) that defines object macros and assumption/conclusion environments, and a Lean 4 Renderer that walks the parsed document tree and emits `.lean` files. A companion `.sty` file handles PDF rendering. The plugin is registered in `plastex.cfg` and via `pyproject.toml` entry points.

**Tech Stack:** Python ≥ 3.9, plasTeX ≥ 3.1, Jinja2, pytest, pdflatex

```python
from plasTeX import Environment


class assumptions(Environment):
    r"""\begin{assumptions} ... \end{assumptions}"""

    def lean4_items(self) -> list[str]:
        """Return Lean 4 strings for each \item in this environment."""
        results = []
        for item in self.childNodes:
            if getattr(item, 'nodeName', '') != '@item':
                continue
            # Find the first macro child with a lean4() method
            macro = self._first_ff_macro(item)
            if macro is not None:
                results.append(macro.lean4())
            else:
                text = item.textContent.strip()
                results.append(f'-- TODO: formalise assumption: {text}')
        return results

    def _first_ff_macro(self, item):
        for child in item.childNodes:
            if callable(getattr(child, 'lean4', None)):
                return child
        return None


class conclusion(Environment):
    r"""\begin{conclusion} ... \end{conclusion}"""

    def lean4_return(self) -> str:
        """Return the Lean 4 return type for this conclusion."""
        for child in self.childNodes:
            if callable(getattr(child, 'lean4', None)):
                # conclusion macros return only the predicate, not the full assumption
                return self._lean4_return_type(child)
        text = self.textContent.strip()
        return f'True /- sketch: {text} -/'

    def _lean4_return_type(self, macro) -> str:
        """Extract the return type (not the full assumption binding) from a macro."""
        if macro.nodeName == 'unit':
            elem = macro.attributes['elem'].textContent.strip()
            return f'IsUnit ({elem})'
        # Fallback: use lean4() and strip any leading binding
        return macro.lean4()
```

```python
# src/formalization_flow/Renderers/lean4/__init__.py
from __future__ import annotations
from pathlib import Path
from plasTeX.Renderers import Renderer as BaseRenderer


class Renderer(BaseRenderer):
    """plasTeX renderer that emits sorry-filled Lean 4 theorem stubs."""

    def render(self, document) -> None:
        output = self.render_to_string(document)
        jobname = document.userdata.get('jobname', 'output')
        Path(f"{jobname}.lean").write_text(output, encoding='utf-8')

    def render_to_string(self, document) -> str:
        parts: list[str] = []
        self._walk(document, parts)
        return ''.join(parts)

    def _walk(self, node, parts: list[str]) -> None:
        for child in getattr(node, 'childNodes', []):
            name = getattr(child, 'nodeName', '')
            if name in ('theorem', 'lemma'):
                parts.append(self._render_theorem(child))
            else:
                self._walk(child, parts)

    def _render_theorem(self, node) -> str:
        label = self._get_label(node)
        if not label:
            raise ValueError(
                r"Theorem/lemma has no \label. "
                r"Every theorem and lemma must have a \label{...}."
            )

        assumptions_node = self._find_child(node, 'assumptions')
        conclusion_node = self._find_child(node, 'conclusion')

        params = ''
        if assumptions_node is not None:
            items = assumptions_node.lean4_items()
            todo_comments = [i for i in items if i.startswith('-- TODO')]
            lean_params = [i for i in items if not i.startswith('-- TODO')]
            prefix = '\n'.join(todo_comments) + '\n' if todo_comments else ''
            params = ' '.join(lean_params)
        else:
            prefix = ''

        ret = 'True'
        if conclusion_node is not None:
            ret = conclusion_node.lean4_return()

        return (
            f"{prefix}"
            f"theorem {label} {params} :\n"
            f"    {ret} := by\n"
            f"  sorry\n\n"
        )

    def _get_label(self, node) -> str | None:
        label = node.attributes.get('label') if node.attributes else None
        if label:
            return str(label).strip()
        # Also check userdata (some plasTeX versions store label there)
        return node.userdata.get('label')

    def _find_child(self, node, nodename: str):
        for child in getattr(node, 'childNodes', []):
            if getattr(child, 'nodeName', '') == nodename:
                return child
        return None
```

## Known limitations (alpha)

- HTML templates for object macros may need tuning once rendered in-browser — the Jinja2 attribute access pattern (`self.attributes.name`) may need adjustment depending on the plasTeX version.
- The `\label` extraction mechanism (`_get_label`) may need adjustment depending on how the installed plasTeX version exposes labels on theorem nodes.
- The `@item` nodeName for list items inside `assumptions` may differ — check with a quick print of `child.nodeName` if environment tests fail.
- Lean 4 output file path is determined by plasTeX's output config; may need a `plastex.cfg` entry to control it.
