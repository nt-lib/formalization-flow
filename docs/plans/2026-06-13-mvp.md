# FormalizationFlow MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A plasTeX plugin that lets a mathematician write one `.tex` file and build it as PDF (pdflatex), HTML (plastex), and sorry-filled Lean 4 (plastex --renderer lean4).

**Architecture:** The Python package `formalization_flow` acts as a plasTeX plugin with three rendering layers, each driven by its own files:
- **LaTeX/PDF**: `sty/formalizationflow.sty` — standard `\newcommand` definitions.
- **HTML**: `renderer_templates/html5/*.jinja2` — Jinja2 templates registered into the HTML5 renderer via `ProcessOptions`.
- **Lean 4**: `renderer_templates/lean4/assumption/*.jinja2` and `renderer_templates/lean4/conclusion/*.jinja2` — Jinja2 templates loaded by the Lean 4 renderer at render time.

Command classes in `formalizationflow.py` are thin shells that only declare argument structure (`args`). No rendering logic lives in Python — it all lives in template files. Adding a new mathematical object means dropping template files in the right directories, with no Python required.

**Tech Stack:** Python ≥ 3.9, plasTeX ≥ 3.1, Jinja2, pytest, pdflatex

```python
# src/formalization_flow/Renderers/lean4/__init__.py
from __future__ import annotations
from pathlib import Path
import jinja2
from plasTeX.Renderers import Renderer as BaseRenderer

TEMPLATE_DIR = (
    Path(__file__).parent.parent.parent
    / 'Packages' / 'renderer_templates' / 'lean4'
)

_jinja_env = jinja2.Environment(
    loader=jinja2.FileSystemLoader([
        str(TEMPLATE_DIR / 'assumption'),
        str(TEMPLATE_DIR / 'conclusion'),
    ]),
    undefined=jinja2.Undefined,
)


def _node_attrs(node) -> dict[str, str]:
    """Extract node attributes as plain strings."""
    return {
        k: v.textContent.strip()
        for k, v in (node.attributes or {}).items()
    }


def _render_assumption_macro(node) -> str | None:
    """Render a macro node using its lean4/assumption template.
    Returns None if no template exists for this macro."""
    try:
        tpl = _jinja_env.get_template(f'assumption/{node.nodeName}.jinja2')
        return tpl.render(**_node_attrs(node)).strip()
    except jinja2.TemplateNotFound:
        return None


def _render_conclusion_macro(node) -> str | None:
    """Render a macro node using its lean4/conclusion template.
    Returns None if no template exists for this macro."""
    try:
        tpl = _jinja_env.get_template(f'conclusion/{node.nodeName}.jinja2')
        return tpl.render(**_node_attrs(node)).strip()
    except jinja2.TemplateNotFound:
        return None


class Renderer(BaseRenderer):
    """plasTeX renderer that emits sorry-filled Lean 4 theorem stubs."""

    def render(self, document) -> None:
        output = self.render_to_string(document)
        jobname = document.userdata.get('jobname', 'output')
        Path(f'{jobname}.lean').write_text(output, encoding='utf-8')

    def render_to_string(self, document) -> str:
        parts: list[str] = []
        self._walk(document, parts)
        return ''.join(parts)

    def _walk(self, node, parts: list[str]) -> None:
        for child in getattr(node, 'childNodes', []):
            if getattr(child, 'nodeName', '') in ('theorem', 'lemma'):
                parts.append(self._render_theorem(child))
            else:
                self._walk(child, parts)

    def _render_theorem(self, node) -> str:
        label = self._get_label(node)
        if not label:
            raise ValueError(
                r'Theorem/lemma has no \label. '
                r'Every theorem and lemma must have a \label{...}.'
            )

        params, todo_comments = self._render_assumptions(node)
        ret = self._render_conclusion(node)

        prefix = '\n'.join(todo_comments) + '\n' if todo_comments else ''
        param_str = ' '.join(params)

        return (
            f'{prefix}'
            f'theorem {label} {param_str} :\n'
            f'    {ret} := by\n'
            f'  sorry\n\n'
        )

    def _render_assumptions(self, theorem_node) -> tuple[list[str], list[str]]:
        """Return (lean_params, todo_comments) from the assumptions environment."""
        asmp = self._find_child(theorem_node, 'assumptions')
        if asmp is None:
            return [], []

        lean_params: list[str] = []
        todo_comments: list[str] = []

        for item in asmp.childNodes:
            if getattr(item, 'nodeName', '') != '@item':
                continue
            macro = self._first_macro_child(item)
            if macro is not None:
                rendered = _render_assumption_macro(macro)
                if rendered is not None:
                    lean_params.append(rendered)
                else:
                    todo_comments.append(
                        f'-- TODO: no lean4 assumption template for \\{macro.nodeName}'
                    )
            else:
                text = item.textContent.strip()
                todo_comments.append(f'-- TODO: formalise assumption: {text}')

        return lean_params, todo_comments

    def _render_conclusion(self, theorem_node) -> str:
        conc = self._find_child(theorem_node, 'conclusion')
        if conc is None:
            return 'True /- no conclusion -/'

        macro = self._first_macro_child(conc)
        if macro is not None:
            rendered = _render_conclusion_macro(macro)
            if rendered is not None:
                return rendered
            # No conclusion template — fall back to a comment
            return f'True /- no lean4 conclusion template for \\{macro.nodeName} -/'

        text = conc.textContent.strip()
        return f'True /- sketch: {text} -/'

    def _get_label(self, node) -> str | None:
        label = (node.attributes or {}).get('label')
        if label:
            return str(label).strip()
        return node.userdata.get('label')

    def _find_child(self, node, nodename: str):
        for child in getattr(node, 'childNodes', []):
            if getattr(child, 'nodeName', '') == nodename:
                return child
        return None

    def _first_macro_child(self, item):
        """Return the first child node that has a known Lean 4 template."""
        for child in getattr(item, 'childNodes', []):
            name = getattr(child, 'nodeName', '')
            if (TEMPLATE_DIR / 'assumption' / f'{name}.jinja2').exists():
                return child
        return None
```

## Extending with a new mathematical object (user guide)

To add `\group{G}` mapping to `(G : Type*) [Group G]`:

1. Add `\newcommand{\group}[1]{Let $#1$ be a group.}` to `sty/formalizationflow.sty`.
2. Add `src/formalization_flow/Packages/renderer_templates/html5/group.jinja2`:
   ```
   Let <i>{{ self.attributes.name }}</i> be a group.
   ```
3. Add `src/formalization_flow/Packages/renderer_templates/lean4/assumption/group.jinja2`:
   ```
   ({{ name }} : Type*) [Group {{ name }}]
   ```
4. Add a `group` Command class to `formalizationflow.py` with `args = 'name'`.

No other Python changes required.

## Known limitations (alpha)

- HTML templates (`self.attributes.name`) may need adjustment depending on how the installed plasTeX version exposes attribute nodelists in Jinja2 context.
- The `@item` nodeName for list items inside `assumptions` may differ across plasTeX versions.
- The `\label` extraction mechanism may need adjustment — check `thm.attributes` and `thm.userdata` if labels are not found.
- Lean 4 output file path is determined by plasTeX's output config and may need a `plastex.cfg` entry to control it.
- The Jinja2 `FileSystemLoader` in the renderer points to the installed package path. If running from source without `pip install -e .`, the template path may need adjustment.
